Database Design

User: {
  id: string,
  username: string,
  password: string,
  profile: string,                    // Profile id
  history: string,                    // history id
  dailyStreakLastUpdateTime: Date,
  dailyStreakCount: Integer,
  longestStreak: Integer,
  lessons: string[]                   // Lesson ids
}

Profile: {
  id: string,
  bgColor: string,                    // hex code
  baseImage: string,                  // image url (ex canadian goose, the white goose, the cute brown one)
  accessory: string,                  // image url 
  heldItem: string,                   // image url (knife, teddy bear, SHARK STUFFY WITH EYES CLOSED)
}

Lesson: {
  id: string,
  name: string,
  author: string                      // userId
  lives: Integer,
  livesLastZeroTime: Date,            // the timestamp of when the user's lives hit zero
  streakCount: Integer,
  pageProgress: Integer,              // the furthest page the user has progressed to
  highScore: Integer,
  currentScore: Integer,              // the current score of the user as they progress through the quiz score. 
  info: string,                       // Info id
  chapters: string[]                  // Chapter ids
}

Info: {
  id: string,
  title: string,
  content: string[],
  fact: string
}

Chapter: {
  id: string,
  name: string,
  info: string,                       // Info id
  questions: string[]                 // Question ids
}

Question: {
    id: string,
    type: string,                     // Is one of: "DragAndDrop", "MultipleChoice", or "Matching"
    ...                               // additional fields depending on type. Refer to classes in frontend/src/class for fields
}

LessonConfig: {
  livesPerLesson: Integer,
  livesRefreshCooldown: Integer,      // cooldown period, seconds
  scoreMultiplier: Integer,           // Either 100 or 1000. scoreMultiplier * (streakCount/2 + 1) = points for each question 
}

profilePictureConfig: {
  baseImage: string[],                 // image url (ex canadian goose, the white goose, the cute brown one)
  accessory: string[],                 // image url 
  heldItem: string[],                  // image url (knife, teddy bear, SHARK STUFFY WITH EYES CLOSED)
}

GlobalHistory: {
  lessons: string[],                  // lesson of the day ids history
}

UserHistory: {
  id: string,
  lessonsPerDay: Integer[],           // number of lessons per day, array of length 7. arr[0] is sunday, arr[1], ... this is to display in user stats
}
